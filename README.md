import java.util.*;
  
  public class Main{
  
  public static void main(String[] args) {
      Scanner sc = new Scanner(System.in);
      
      int n = sc.nextInt();
      int arr[] = new int[n];
      for(int i = 0 ; i < n ; i++){
          arr[i] = sc.nextInt();
      }
      for(int i = 0 ; i < n ; i++){
          
          if(arr[i] == 2 || arr[i] == 0){
               System.out.println("not prime");
          }
       else if (arr[i] % 2 == 0 || arr[i] % 3 == 0 ){
           System.out.println("not prime");
       }
       else
       {
           System.out.println("prime");
       }
       
      }
   }
  }
